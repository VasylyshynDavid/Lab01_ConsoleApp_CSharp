using System;

public class Program
{
    public static void Main()
    {
        // Створюємо нескінченний цикл, як у віконній програмі
        while (true)
        {
            try
            {
                // --- Імітація введення ---
                Console.WriteLine("\n--- Новий розрахунок (Варіант 4) ---");
                Console.Write("Введіть X1 (або 'вихід' для завершення): ");
                string input1 = Console.ReadLine();

                // Імітація кнопки "Вихід"
                if (input1.ToLower() == "вихід")
                {
                    break; // Виходимо з циклу
                }

                Console.Write("Введіть X2: ");
                string input2 = Console.ReadLine();

                double x1 = double.Parse(input1);
                double x2 = double.Parse(input2);

                // --- Імітація кнопки "Обчислити" ---
                
                // 1. Основна функція: y = sin(x1 - x2^3 + sqrt(x1)) - 1.3 * x1^3
                double y = Math.Sin(x1 - Math.Pow(x2, 3) + Math.Sqrt(x1)) - 1.3 * Math.Pow(x1, 3);

                // 2. Додаткове завдання: менше із значень X1, X2
                double minVal = Math.Min(x1, x2);

                // 3. Форматування: 3 знаки після коми
                string format = "0.000";

                // --- Виведення результатів ---
                Console.WriteLine("\n--- Результат розрахунку ---");
                Console.WriteLine($"Результат Y: {y.ToString(format)}");
                Console.WriteLine($"Додатково (менше з X1, X2): {minVal.ToString(format)}");
                
                // Імітація кнопки "Очистити" відбувається 
                // автоматично при наступній ітерації циклу
            }
            catch (FormatException)
            {
                // Спрацює, якщо введено не число
                Console.WriteLine("Помилка! Будь ласка, вводьте тільки числа.");
            }
            catch (Exception ex)
            {
                // Для інших помилок (наприклад, корінь з від'ємного X1)
                Console.WriteLine($"Помилка обчислень: {ex.Message}");
            }
        }

        Console.WriteLine("Роботу завершено.");
    }
}
